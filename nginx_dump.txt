  listen 80;
  listen [::]:80;
  server_name wiki.bit-habit.com www.wiki.bit-habit.com;
  include /etc/nginx/snippets/security-headers.conf;
  include /etc/nginx/snippets/security-block.conf;
  include /etc/nginx/snippets/rate-limit.conf;
  return 301 https://wiki.bit-habit.com$request_uri;
}
server {
  listen 443 ssl http2;
  listen [::]:443 ssl http2;
  server_name wiki.bit-habit.com;
  include /etc/nginx/snippets/security-headers.conf;
  include /etc/nginx/snippets/security-headers-ssl.conf;
  include /etc/nginx/snippets/security-block.conf;
  include /etc/nginx/snippets/rate-limit.conf;

  ssl_certificate           /etc/letsencrypt/live/bit-habit.com-0001/fullchain.pem;
  ssl_certificate_key       /etc/letsencrypt/live/bit-habit.com-0001/privkey.pem;
  ssl_trusted_certificate   /etc/letsencrypt/live/bit-habit.com-0001/fullchain.pem;
  include /etc/letsencrypt/options-ssl-nginx.conf;
  ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

  ssl_stapling on;
  ssl_stapling_verify on;
  resolver 1.1.1.1 8.8.8.8 valid=300s;
  resolver_timeout 5s;

  location / {
    proxy_pass http://127.0.0.1:3001;
    proxy_http_version 1.1;
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
  }
}
server {
  listen 443 ssl http2;
  listen [::]:443 ssl http2;
  server_name www.wiki.bit-habit.com;
  include /etc/nginx/snippets/security-headers.conf;
  include /etc/nginx/snippets/security-headers-ssl.conf;
  include /etc/nginx/snippets/security-block.conf;
  include /etc/nginx/snippets/rate-limit.conf;

  ssl_certificate           /etc/letsencrypt/live/bit-habit.com-0001/fullchain.pem;
  ssl_certificate_key       /etc/letsencrypt/live/bit-habit.com-0001/privkey.pem;
  ssl_trusted_certificate   /etc/letsencrypt/live/bit-habit.com-0001/fullchain.pem;
  include /etc/letsencrypt/options-ssl-nginx.conf;
  ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

  return 301 https://wiki.bit-habit.com$request_uri;
}
